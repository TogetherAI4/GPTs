# Maze generation using randomized Prim's algorithm
def generate_maze(width, height):
    maze = np.zeros((height, width), dtype=np.int8)
    directions = [(0, 1), (1, 0), (0, -1), (-1, 0)]
    start = (1, 1)
    maze[start] = 1
    walls = [(start[0] + d[0], start[1] + d[1]) for d in directions]
    while walls:
        wall = walls.pop(np.random.randint(len(walls)))
        if 0 < wall[0] < height - 1 and 0 < wall[1] < width - 1:
            if maze[wall] == 0:
                opposite = (wall[0] + (wall[0] - start[0]), wall[1] + (wall[1] - start[1]))
                if maze[opposite] == 1:
                    maze[wall] = 1
                    maze[(start[0] + wall[0]) // 2, (start[1] + wall[1]) // 2] = 1
                    start = wall
                    for d in directions:
                        new_wall = (wall[0] + d[0], wall[1] + d[1])
                        if maze[new_wall] == 0:
                            walls.append(new_wall)
    return maze

# A* algorithm for pathfinding
def astar(maze, start, end):
    directions = [(0, 1), (1, 0), (0, -1), (-1, 0)]
    g_score = {start: 0}
    f_score = {start: np.linalg.norm(np.array(start) - np.array(end))}
    open_set = [(f_score[start], start)]
    came_from = {}

    while open_set:
        current = heapq.heappop(open_set)[1]
        if current == end:
            path = []
            while current in came_from:
                path.append(current)
                current = came_from[current]
            return path[::-1]
        for d in directions:
            neighbor = (current[0] + d[0], current[1] + d[1])
            if 0 <= neighbor[0] < maze.shape[0] and 0 <= neighbor[1] < maze.shape[1]:
                if maze[neighbor] == 1:
                    tentative_g_score = g_score[current] + 1
                    if tentative_g_score < g_score.get(neighbor, float('inf')):

                        g_score[neighbor] = tentative_g_score
                        f_score[neighbor] = tentative_g_score + np.linalg.norm(np.array(neighbor) - np.array(end))
                        heapq.heappush(open_set, (f_score[neighbor], neighbor))
    return None
